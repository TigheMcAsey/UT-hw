        \textbf{(i)} Since specifying a point on a line through the origin determines the line, we may write
        \begin{align*}
            U_0 &\to \pi^{-1}(U_0) \\
            [z_0,z_1] &\mapsto ([z_0,z_1],(1,z_1/z_0))
        \end{align*}
        abbreviated as \(\sigma_0: ([z_0,z_1]) \mapsto (1,w_0)\), similarly for \(\sigma_1\) on \(U_1\). That these define a section is tautological, since \(\pi\circ \sigma_j = 1_{U_j}\), and \(w_j\) are the coordinate functions on \(U_j\), so that in coordinates \(\sigma_j\) is given by \(w_j \mapsto (w_j,w_j)\) is holomorphic. To identify \(U_j \times \mathbb{C}\) with \(\pi^{-1}(U_j)\), simply take \(([z_0,z_1],\lambda) \leftrightarrow \lambda\sigma_0([z_0,z_1]) := ([z_0,z_1],\lambda(1, w_0))\) and analogously on \(U_1\). These identifications are holomorphic since multiplication is and the \(\sigma_j\) are.

        \textbf{(ii)} We define in \(U_0\) coordinates \(\rho_{01}: w_1 \mapsto w_0 = \frac{1}{w_1}\), this is trivially holomorphic on \(U_0 \cap U_1\) and since \(w_0\cdot w_1 = 1\), we get that
        \begin{align*}
            \rho_{01}\sigma_1 = \rho_{01}(1,w_1) = w_0\cdot(1,w_1) = (w_0,1) = \sigma_0
        \end{align*}

        \textbf{(iii)} This is a simple algebraic check using part (ii).
        \begin{align*}
            f_0 \sigma_0 = f_1 \sigma_1 \implies f_0 \sigma_1 =  \rho_{01}^{-1}\rho_{01}f_0 \sigma_1 = \rho_{01}^{-1}f_0 \sigma_0 = \rho_{01}^{-1}f_1 \sigma_1
        \end{align*}
        So that \(f_0 = \rho_{01}^{-1}\sigma_1\)

        \textbf{(iv)}
        \begin{align*}
            \gen{\sigma_j,\sigma_j} = \gen{(1,w_j),(1,w_j)} = 1+\abs{w_j}^2
        \end{align*}
        Then \(\abs{t_{01}}^2 = \abs{w_0}^2\), so that
        \begin{align*}
            \frac{1}{\abs{t_{01}}^2}h_1 = \abs{w_0}^2(1 + \abs{w_1}^2) = \abs{w_0}^2 + \abs{w_0\cdot w_1}^2 = \abs{w_0}^2 + 1 = h_0
        \end{align*}

        \textbf{(v)}
        \begin{align}
            -\pwpw \log(h_0) &= -\pwpw \log(1 + \abs{w_0}^2) = -\frac{\partial}{\partial \overline{w_0}}\frac{\overline{w_0}}{1 + \abs{w_0}^2} = -\frac{1}{1 + \abs{w_0}^2} + \frac{\overline{w_0}w_0}{(1 + \abs{w_0}^2)^2} \\
            &= -\frac{1 + \abs{w_0}^2 - \abs{w_0}^2}{(1 + \abs{w_0}^2)^2} = \frac{-1}{(1 + \abs{w_0}^2)^2} \label{curvaturecalc}
        \end{align}

        \textbf{(vi)}
        \begin{align}
            \frac{i}{2\pi}\int_{\mathbb{P}^1} F_{h^{-1}_{FS}}dz \wedge d\bar{z} &=  \frac{i}{2\pi}\int_{\mathbb{P}^1 \setminus \set{[0:1]}} F_{h^{-1}_{FS}}dz \wedge d\bar{z} =  \frac{i}{2\pi}\int_{U_0} F_{h^{-1}_{FS}}dz \wedge d\bar{z} \\ &\overset{(v)}{=} \frac{i}{2\pi}\int_{\mathbb{C}} \frac{-1}{(1 + \abs{w_0}^2)^2}dw_0 \wedge d\overline{w_0} \label{precoordchange}
        \end{align}
        To evaluate the integral in \eqref{precoordchange}, we use polar coordinates, \(w_0 = re^{i\theta} \tand w_1 = re^{-i\theta}\) so that \(dw_0 \wedge dw_1\) is computed as
        \begin{align}
            (e^{i\theta}dr +ire^{i\theta}d\theta)\wedge(e^{-i\theta}dr - ire^{-i\theta}d\theta) = -2ir dr\wedge d\theta
        \end{align}
        Applying this change of coordinates to \eqref{precoordchange} yields
        \begin{align}
            \frac{i}{2\pi}\int_{\mathbb{P}^1} F_{h^{-1}_{FS}}dz \wedge d\bar{z} &= \frac{1}{\pi}\int_0^{2\pi}\int_0^\infty \frac{-r}{(1 + r^2)^2}drd\theta = \frac{1}{\pi}\int_0^{2\pi} \int_1^\infty \frac{du}{u^2}d\theta = \frac{1}{\pi}\int_0^{2\pi} -\frac12d\theta = -1 \label{computeint}
        \end{align}

        \textbf{(vii)} Recall the dual of a bundle satisfies \(L \otimes L^{-1}\) is the trivial bundle. So the transition functions are given by the inverse transition functions, i.e. the transition functions are \(\set{t_{01}, t_{10}} = \set{\rho_{01}^{-1}, \rho_{10}^{-1}}\). To check that \(h_{\text{FS}}\) defines a hermitian metric on \(\mathcal{O}_{\mathbb{P}^1}(1)\), we simply need to check that \(h \in \Gamma(\mathbb{P}^1,\mathcal{O}_{\mathbb{P}^1}(-1)\otimes \overline{\mathcal{O}_{\mathbb{P}^1}(-1)})\), since \(h_{\text{FS}}\) is already defined to be real and positive globally. The transition functions on \(\mathcal{O}_{\mathbb{P}^1}(-1)\otimes \overline{\mathcal{O}_{\mathbb{P}^1}(-1)}\) are given by \(e_{01} = t_{01}\overline{t_{01}} = \abs{t_{01}}^2\) and \(e_{10} = t_{10} \overline{t_{10}} = \abs{t_{10}}^2\), so that from (iv), we have \(h_0 = \abs{t_{01}}^{-2}h_1\) on \(U_0 \cap U_1\) whence \(h_{\text{FS}} = h_0^{-1} = \abs{t_{01}}^2h_1^{-1}\) on \(U_1 \cap U_2\) satisfies the first transition law, and using that \(t_{10} = t_{01}^{-1}\), we readily see that \(h_{\text{FS}}\) satisfies both transition laws, so it is indeed a section of \(\mathcal{O}_{\mathbb{P}^1}(-1)\otimes \overline{\mathcal{O}_{\mathbb{P}^1}(-1)}\), and thus a hermitian metric.

        \textbf{(viii)} The \(\ell_j\) are smooth by definition in either chart, we need only show that the \(\ell_j\) satisfy the transition laws. It suffices to check for just \(t_{01}\), where we have \(t_{01}(w_1) = t_{01}(w_0^{-1}) = \rho_{01}(w_0^{-1})^{-1} = (w_0^{-1})^{-1} = w_0\), so that \(t_{01}\ell_1(w_1) = w_0\ell_1(w_1) = w_0\cdot w_1 = 1\). Similarly for \(\ell_2\), we get \(t_{01}\ell_2(w_1) = t_{01}\cdot 1 = w_0\).

        \textbf{(ix)} \(F_{h_{\text{FS}}} \overset{\text{def}}{=} -\pzpz \log h_{\text{FS}}\), on \(U_0\) we have \(h_{\text{FS}} = h_0^{-1}\), so that on \(U_0\) we get locally
        \begin{align}
            F_{h_{\text{FS}}} = -\pzpz \log h_{\text{FS}} = -\pzpz \log h_0^{-1} = \pzpz \log h_0 \overset{\eqref{curvaturecalc}}{=} \frac{1}{(1 + \abs{w_0}^2)^2}
        \end{align}
        Using this expression, and that \(\mathbb{P}^1 \setminus U_0\) has measure zero, we use that the expression of \(F_{h_{\text{FS}}}\) differs from \(F_{h^{-1}_{\text{FS}}}\) to compute
        \begin{align}
            \frac{i}{2\pi}\int_{\mathbb{P}^1}F_{h_{\text{FS}}}dz\wedge d\bar{z} &= \frac{i}{2\pi}\int_{U_0}F_{h_{\text{FS}}}dz\wedge d\bar{z} = \frac{i}{2\pi}\int_{U_0}\frac{1}{(1 + \abs{w_0}^2)^2}dw_0\wedge d\overline{w_0} \\
            &\overset{\eqref{computeint}}{=} 1
        \end{align}

        \textbf{(x)} Since the integral of \(\frac{i}{2\pi}\) times the curvature over a Riemann surface gives the number of zeroes - number of poles of an arbitrary nonzero meromorphic section of its line bundle, we know that the holomorphic sections of \(\mathcal{O}_{\mathbb{P}_1}(1)\) have exactly one zero. In order to vanish once, it must be linear in one of the charts \(\set{U_0,U_1}\), then on \(\mathbb{P}^1\), a holomorphic section is determined by its behaviour on one of the charts, since at \(w \not \in U_j\), we have \(\varphi(w) = \lim_{n \to \infty} \varphi(w_n)\) for \(w_n \in U_0 \cap U_1\). Thus letting \(\varphi\) be a holomorphic section of \(\mathcal{O}_{\mathbb{P}^1}(1)\), we find that \(\varphi\) on \(U_0\) is given by \(aw_0 + b \overset{\text{loc}}{=} a \ell_2 + b \ell_1\) but since this is equal to \(a \ell_2 + b\ell_1\) on \(U_0\), and \(\varphi\) satisfies the same transition functions, we find they are equal on \(U_1 \setminus \set{[1:0]}\) and thus equal on \(\mathbb{P}^1\) by the earlier discussion. This suffices to show that \(\ell_1,\ell_2\) span \(H^0(\mathbb{P}^1,\mathcal{O}_{\mathbb{P}^1}(1))\).

        \textbf{(i)} The covering map \(\rho: \mathbb{C} \to X\) induces coordinates, so long as it is injective, we define charts for which it is injective. Define the following charts projecting to an open cover.
        \begin{align*}
            U_1 &= \set{z \mid 0 < \Im(z) < \Im(\tau) \tand 0 < \Re(z) < 1} \\
            U_2 &= \set{z \mid 0 < \Im(z) < \Im(\tau) \tand 1/2 < \Re(z) < 3/2} \\
            U_3 &= \set{z \mid \Im(\tau)/2 < \Im(z) < 3\Im(\tau)/2 \tand 0 < \Re(z) < 1} \\
            U_4 &= \set{z \mid \Im(\tau)/2 < \Im(z) < 3\Im(\tau)/2 \tand 1/2 < \Re(z) < 3/2}
        \end{align*}
        Then let \(s_j\) be the lift (equiv. coordinate chart) of \(\rho\) for each \(U_j\), \(\theta_1\) is well defined on the torus up to choice of section, so we get \(\theta_1(z_j) = \theta_1(s(z_j))\), all of the \(dz_j/dz_i = 1\), so it suffices to choose \(t_{ij}\) such that \(\theta_1(z_j) = t_{ji}\theta_1(z_i)\). Define
        \begin{align*}
            &t_{j+1,j}(z) = \begin{cases}
                1 & \Re(z_j) \in (1/2,1) \\
                -1 & \Re(z_j) \in (0,1/2)
            \end{cases}, \quad j \in \set{1,3} \\
            &t_{j+2,j}(z) = \begin{cases}
                1 & \Im(z_j) \in (\Im(\tau)/2,\Im(\tau)) \\
                \exp(-i\pi \tau - 2\pi i z) & \Im(z_j) \in (0,\frac12)
            \end{cases}, \quad j \in \set{1,2}
        \end{align*}
        The remaining transitions can be defined to be given by the cocycle conditions on these transitions. To see that \(\theta_1\) defines a section, note that \(\theta_1(z + 1) = -\theta_1(z)\) and \(\theta_1(z + \tau) = \exp(-i\pi \tau - 2\pi i z)\theta_1(z)\), so that \(\theta_1\) is a section of the line bundle specified by the \(U_j, \tand t_{ij}\).

        \textbf{(ii)} We have computed the Green's function on the torus is
        \begin{align}
            G(z-w) = \log\abs{\frac{\theta_1(z-w)}{\theta_1'(0)}}^2 - \frac{2\pi}{\Im(\tau)}(\Im(z))^2
        \end{align}
        In order to get that we choose \(h = \frac{1}{\abs{\theta'(0)}^2}\exp(-\frac{2\pi}{\Im \tau}(\Im z)^2)\), denoting the line bundle from part (i) as \(L\), we get the transition functions on \(L^{-1}\otimes \overline{L^{-1}}\) are \(1\) between \(U_1 \tand U_2\) and between \(U_3 \tand U_4\), and \(\frac{1}{\abs{e^{-i\pi \tau - 2\pi i z}}^2} = e^{-2\pi \Im(\tau) - 4\pi \Im(z)}\) between \(U_3\) and \(U_1\) and between \(U_4\) and \(U_2\), the rest of the transition laws are determined by the cocycle conditions. \(h\) is defined so that 
        \begin{align}
            \log\abs{\theta_1(z-w)}^2_h = \log\abs{\frac{\theta_1(z-w)}{\theta_1'(0)}}^2 + \log \exp(-\frac{2\pi}{\Im \tau}(\Im z)^2) = \log\abs{\frac{\theta_1(z-w)}{\theta_1'(0)}}^2 - \frac{2\pi}{\Im(\tau)}(\Im(z))^2 = G(z-w)
        \end{align}
        And moreover \(h\) is a hermitian metric since \(h(z) \in \mathbb{R}_{>0}\) for all \(z\), and denoting the transition laws for \(L^{-1}\otimes \overline{L^{-1}}\) as \(\widetilde{t_{ij}}\) we find that \(h(z+1) = h(z) = \widetilde{t}_{j+1,j}h(z)\) for \(j = 1,3\) and for \(j = 1,2\) we get
        \begin{align}
            h(z+\tau) &= \frac{1}{\abs{\theta_1'(0)}^2}\exp(-\frac{2\pi}{\Im \tau}(\Im (z + \tau))^2) = \frac{1}{\abs{\theta_1'(0)}^2}\exp(-\frac{2\pi}{\Im\tau}(\Im z)^2 - 4\pi \Im(z) - 2\pi \Im(\tau)) \\
            &=\exp(-4\pi \Im(z) - 2\pi\Im(\tau))h(z) = \tilde{t}_{j+2,j}h(z)
        \end{align}
        Since the other transition laws are given by the cocycle condition with these transition laws we see that \(h\) is indeed a hermitian metric.

        Now, we compute the curvature to simplify things denote \(C = \frac{1}{\abs{\theta_1'(0)}^2}\) and \(D = -\pi/2\Im(\tau)\) so that \(h(z) = Ce^{D(z-\bar{z})^2}\), it follows we can compute the curvature of \(h\) as follows
        \begin{align}
            -\pzpz \log h &= -\pzpz \log Ce^{D(z-\bar{z})^2} = -\pzpz\left(\log C + D(z - \bar{z})^2\right) = -D\pzpz (z^2 - 2z\bar{z} + \bar{z}^2) \\
            &= -D\pzb (2z - 2\bar{z}) = 2D \overset{\text{def}}{=} -\pi/\Im(\tau)
        \end{align}
        This curvature is special in that it corresponds to one of the key desired properties of the \(\theta_1\) function, namely having a single simple zero on the torus. To see this we let \(s\) be an arbitrary meromorphic section of \(L\), then
        \begin{align}
            \#\set{\text{zeroes of }s} - \#\set{\text{poles of }s} = \frac{i}{2\pi}\int_{\mathbb{C}/\mathbb{Z} + \tau \mathbb{Z}} F_{z\bar{z}}dzd\bar{z} \label{curvaturecounting}
        \end{align}
        Using that the curvature is constant we compute the right hand side of \eqref{curvaturecounting} to be 
        \begin{align}
            \frac{-i}{2\Im(\tau)}\int_{\mathbb{C}/\mathbb{Z} + \tau \mathbb{Z}}dzd\bar{z} &= \frac{-i}{2\Im(\tau)} \int_{\text{Fundamental Domain}} 2i dxdy = \frac{1}{\Im(\tau)}\cdot \text{Area of Fundamental Domain} \\
            &= 1
        \end{align}
        Of course, we know this without carrying out the computation since \(\theta_1\) has a simple zero and no poles. In this sense, the metric corresponding to the Greens function on the torus encodes the information of a section of a line bundle on the torus with a simple pole.

        Let \(\varphi \in C_c^\infty(D)\), and let \(w_1, \hdots, w_n\) be the zeroes and poles of \(f\), we define \(B_\epsilon^j := B_\epsilon(w_j)\), then for all \(\epsilon\) small enough so that each \(B_\epsilon^j\) is disjoint we get
        \begin{align}
            \int_D \pzpz \log\abs{f}^2 \varphi \dv &\overset{\text{def}}{=} \int_D \log\abs{f}^2 \pzpz \varphi \dv \\
            &= \underbrace{\int_{D \setminus \sum B^j_\epsilon} \log\abs{f}^2 \pzpz \varphi \dv}_{(A)} + \underbrace{\int_{\sum B_\epsilon^j} \log\abs{f}^2 \pzpz \varphi \dv}_{(B)} \label{Weakder}
        \end{align}
        Then since \(f: D \to \mathbb{P}^1\), near any of the \(w_j\) we can write it as \(z^k g\) for \(k \in \mathbb{Z}\) and \(g\) nonvanishing and holomorphic on \(B_\epsilon^j\), moreover since \(\varphi\) is \(C^\infty\) and compactly supported, it is bounded on \(D\) along with all its derivatives. It follows that \((B) \in \mathcal{O}(\epsilon^2\log\abs{\epsilon^k g}^2) = \mathcal{O}(\epsilon^2\log(\epsilon))\). Now to evaluate (A), we use Stoke's theorem, we compute on \(D \setminus \sum_1^n B_\epsilon^j\) that \(d \log \abs{f}^2 \pzb \varphi d\bar{z} = \log\abs{f}^2 \pzpz \varphi \dv + \frac{f'}{f}\pzb \varphi \dv\), so that
        \begin{align}
            (A) &\overset{\text{Stoke's}}{=} \int_{\partial (D \setminus \sum B_\epsilon^j)}\log \abs{f}^2 \pzb \varphi d\bar{z} - \int_{D \setminus \sum B_\epsilon^j} \frac{f'}{f}\pzb \varphi \dv \\
            &= -\underbrace{\int_{\sum\partial B_\epsilon^j}\log \abs{f}^2 \pzb \varphi d\bar{z}}_{\mathcal{O}(\epsilon\log \epsilon)} + \underbrace{\int_{\partial D}\log \abs{f}^2 \pzb \varphi d\bar{z}}_{\varphi \in C_c^\infty(D) \implies \pzb\varphi\vert_{\partial D} = 0} - \int_{D \setminus \sum B_\epsilon^j} \frac{f'}{f}\pzb \varphi \dv \\
            &= - \int_{D \setminus \sum B_\epsilon^j} \frac{f'}{f}\pzb \varphi \dv + \mathcal{O}(\epsilon\log \epsilon)
        \end{align}
        Now, we recognize that \(d (-\frac{f'}{f}\varphi dz) = \frac{f'}{f}\pzb \varphi \dv\) on \(D \setminus \sum B_\epsilon^j\), since \(f'/f\) is holomorphic on this set. This gives us
        \begin{align}
            (A) &\overset{\text{Stoke's}}{=} \int_{\partial (D \setminus \sum B_\epsilon^j)}\frac{f'}{f}\varphi dz + \mathcal{O}(\epsilon\log \epsilon) = \underbrace{\int_{\partial D} \frac{f'}{f}\varphi dz}_{\varphi \in C^\infty_c(D) \implies \varphi\vert_{\partial D} = 0} - \int_{\sum \partial B_\epsilon^j} \frac{f'}{f}\varphi dz + \mathcal{O}(\epsilon\log \epsilon) \\
            &= -\int_{\sum \partial B_\epsilon^j} \frac{f'}{f}\varphi + \mathcal{O}(\epsilon \log \epsilon) \label{Aformula}
        \end{align}
        Taken together, \eqref{Weakder} and \eqref{Aformula} give us
        \begin{align}
            \int_D \pzpz \log\abs{f}^2 \varphi \dv &= \lim_{\epsilon \to 0}\left(-\int_{\sum \partial B_\epsilon^j} \frac{f'}{f}\varphi + \mathcal{O}(\epsilon \log \epsilon)\right) \\
            &= \lim_{\epsilon \to 0}-\int_{\sum \partial B_\epsilon^j} \frac{f'}{f}\varphi \\
            &= -2\pi i\left(\sum_{w \in \set{f = 0}} (\operatorname{ord}_wf) \varphi(w) - \sum_{w \in \set{f = \infty}} (\operatorname{ord}_wf) \varphi(w)\right) \label{finalweakderRes}
        \end{align}
        Where the last equality follows since \(\int_{\partial B_\epsilon^j}f'/f = \pm \operatorname{ord}_{w_j}f\) where the sign depends on wether \(f(w_j) = 0\) or \(\infty\). Taken together with \(\varphi \in C^\infty\), we get for any \(\epsilon' > 0\), that for sufficiently small \(\epsilon\) we have \(\norm{\varphi\vert_{B_\epsilon^j} - \varphi(w_j)}_\infty < \epsilon'\), so that
        \begin{align}
            \abs{\int_{B_\epsilon^j} \frac{f'}{f} \varphi dz - \int_{B_\epsilon^j} \frac{f'}{f} \varphi(0)dz} = \abs{\int_{B_\epsilon^j} \frac{f'}{f} (\varphi - \varphi(0)) dz} \leq \epsilon' \int_{B_\epsilon^j} \abs{\frac{f'}{f}}
        \end{align}
        But \(f'/f\) has a simple pole at \(w_j\), so that \(\epsilon' \int_{B_\epsilon^j} \abs{\frac{f'}{f}} \in \mathcal{O}(\epsilon')\), thus goes to zero in the limit. Now since \(\varphi\) was arbitrary, (10) tells us exactly that
        \begin{align}
            \frac{i}{2\pi}\pzpz \log\abs{f}^2 \varphi \dv = \sum_{w \in \set{f = 0}}(\operatorname{ord}_wf) \varphi(w) - \sum_{w \in \set{f = \infty}} (\operatorname{ord}_wf) \varphi(w)
        \end{align}
